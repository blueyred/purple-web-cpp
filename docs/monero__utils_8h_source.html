166  static void free(std::shared_ptr<monero_block> block) {
167  for (std::shared_ptr<monero_tx>& tx : block->m_txs) {
168  tx->m_block.reset();
169  monero_tx_wallet* tx_wallet = dynamic_cast<monero_tx_wallet*>(tx.get());
170  if (tx_wallet != nullptr) {
171  if (tx_wallet->m_outgoing_transfer != boost::none) tx_wallet->m_outgoing_transfer.get()->m_tx.reset();
172  for (std::shared_ptr<monero_transfer> transfer : tx_wallet->m_incoming_transfers) transfer->m_tx.reset();
173  for (std::shared_ptr<monero_output> output : tx_wallet->m_outputs) output->m_tx.reset();
174  for (std::shared_ptr<monero_output> input : tx_wallet->m_inputs) {
175  input->m_key_image.reset();
176  input->m_tx.reset();
177  }
178  }
179  }
180  block.reset();
181  }
